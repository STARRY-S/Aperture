158void process_input(GLFWwindow *window)
159{
160 float speed = 1.0f;

162 speed *= 2.0f;
163 }
164

166 ap_creature_process_move(AP_DIRECTION_FORWARD, speed);
167 }

169 ap_creature_process_move(AP_DIRECTION_BACKWORD, speed);
170 }
171 if (glfwGetKey(window, GLFW_KEY_A) == GLFW_PRESS) {
172 ap_creature_process_move(AP_DIRECTION_LEFT, speed);
173 }
174 if (glfwGetKey(window, GLFW_KEY_D) == GLFW_PRESS) {
175 ap_creature_process_move(AP_DIRECTION_RIGHT, speed);
176 }
177 if (glfwGetKey(window, GLFW_KEY_LEFT_SHIFT) == GLFW_PRESS) {
178 ap_creature_process_move(AP_DIRECTION_DOWN, speed);
179 }
180 if (glfwGetKey(window, GLFW_KEY_SPACE) == GLFW_PRESS) {
181 // ap_creature_process_move(AP_DIRECTION_UP, speed);
182 ap_creature_jump();
183 }
