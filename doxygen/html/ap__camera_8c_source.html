18int ap_camera_generate(unsigned int *camera_id)
19{
20 // initialize camera vector when first use
21 if (camera_vector.data == NULL) {
22 ap_vector_init(&camera_vector, AP_VECTOR_CAMERA);
23 }
24
25 struct AP_Camera camera;
26 ap_camera_init_ptr(&camera);
27 camera.id = camera_vector.length + 1;
28 ap_vector_push_back(&camera_vector, (const char*) &camera);
29 *camera_id = camera.id;
30
31 return AP_ERROR_SUCCESS;
32}

34int ap_camera_use(unsigned int camera_id)
35{
36 if (camera_id > camera_vector.length) {
37 return AP_ERROR_INVALID_PARAMETER;
38 }
39
40 if (camera_id == 0) {
41 camera_using = NULL;
42 return 0;
43 }
44
45 struct AP_Camera *tmp_cam = (struct AP_Camera*) camera_vector.data;
46 for (int i = 0; i < camera_vector.length; ++i) {
47 if (tmp_cam[i].id == camera_id) {
48 camera_using = tmp_cam + i;
49 return 0;
50 }
51 }
52
53 LOGW("failed to use camera: id %u not found", camera_id);
54 return 0;
55}

89int ap_camera_free(int id)
90{
91 if (id <= 0) {
92 return AP_ERROR_INVALID_PARAMETER;
93 }
94
95 struct AP_Camera *data = (struct AP_Camera *) camera_vector.data;
96 for (int i = 0; i < camera_vector.length; ++i) {
97 if (data[i].id != id) {
98 continue;
99 }
100 if (data + i == camera_using) {
101 camera_using = NULL;
102 }
103 ap_vector_remove_data(
104 &camera_vector,
105 (char*) (data + i),
106 (char*) (data + i + 1),
107 sizeof(struct AP_Camera)
108 );
109 break;
110 }
111
112 return 0;
113}

123int ap_camera_update_vectors()
124{
125 if (camera_using == NULL) {
126 return AP_ERROR_CAMERA_NOT_SET;
127 }
128
129 float sin_yaw = (float) sin((double) glm_rad(camera_using->yaw));
130 float cos_yaw = (float) cos((double) glm_rad(camera_using->yaw));
131 float sin_pitch = (float) sin((double) glm_rad(camera_using->pitch));
132 float cos_pitch = (float) cos((double) glm_rad(camera_using->pitch));
133
134 camera_using->front[0] = cos_yaw * cos_pitch;
135 camera_using->front[1] = sin_pitch;
136 camera_using->front[2] = sin_yaw * cos_pitch;
137 glm_vec3_normalize(camera_using->front);
138 return 0;
139}

209int ap_camera_process_movement(int direction, int speed_up)
210{
211 if (camera_using == NULL) {
212 return AP_ERROR_CAMERA_NOT_SET;
213 }
214
215 float dt = 0.0f;
216 ap_render_get_dt(&dt);
217 float velocity = camera_using->speed * dt * speed_up;
218 vec3 temp = { 0.0f, 0.0f, 0.0f };
219 switch (direction)
220 {
221 case AP_DIRECTION_FORWARD:
222 {
223 glm_vec3_scale(camera_using->front, velocity, temp);
224 glm_vec3_add(
225 camera_using->position,
226 temp,
227 camera_using->position
228 );
229 break;
230 }
231 case AP_DIRECTION_BACKWORD:
232 {
233 glm_vec3_scale(camera_using->front, velocity, temp);
234 glm_vec3_sub(
235 camera_using->position,
236 temp,
237 camera_using->position
238 );
239 break;
240 }
241 case AP_DIRECTION_LEFT:
242 {
243 glm_vec3_cross(camera_using->front, camera_using->up, temp);
244 glm_vec3_normalize(temp);
245 glm_vec3_scale(temp, velocity, temp);
246 glm_vec3_sub(
247 camera_using->position,
248 temp,
249 camera_using->position
250 );
251 break;
252 }
253 case AP_DIRECTION_RIGHT:
254 {
255 glm_vec3_cross(camera_using->front, camera_using->up, temp);
256 glm_vec3_normalize(temp);
257 glm_vec3_scale(temp, velocity, temp);
258 glm_vec3_add(
259 camera_using->position,
260 temp,
261 camera_using->position
262 );
263 break;
264 }
265 case AP_DIRECTION_UP:
266 {
267 glm_vec3_scale(camera_using->up, velocity, temp);
268 glm_vec3_add(
269 camera_using->position,
270 temp,
271 camera_using->position
272 );
273 break;
274 }
275 case AP_DIRECTION_DOWN:
276 {
277 glm_vec3_scale(camera_using->up, velocity, temp);
278 glm_vec3_sub(
279 camera_using->position,
280 temp,
281 camera_using->position
282 );
283 break;
284 }
285 default:
286 break;
287 }
288 return 0;
289}

291int ap_camera_process_mouse_move(float x_offset, float y_offset, bool pitch)
292{
293 if (camera_using == NULL) {
294 return AP_ERROR_CAMERA_NOT_SET;
295 }
296 x_offset *= camera_using->sensitivity;
297 y_offset *= camera_using->sensitivity;
298
299 camera_using->yaw += x_offset;
300 camera_using->pitch += y_offset;
301
302 if (pitch) {
303 if(camera_using->pitch > 89.0f) {
304 camera_using->pitch = 89.0f;
305 }
306 if(camera_using->pitch < -89.0f) {
307 camera_using->pitch = -89.0f;
308 }
309 }
310
311 ap_camera_update_vectors();
312 return 0;
313}
