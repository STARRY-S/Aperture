64int ap_light_generate_point(
65 unsigned int *light_id,
66 float position[3],
67 float ambient[3],
68 float diffuse[3],
69 float specular[3],
70 float param[AP_LIGHT_PARAM_NUM])
71{
72 if (!light_id || !param) {
73 return AP_ERROR_INVALID_PARAMETER;
74 }
75
76 if (!point_light_vector.data) {
77 ap_vector_init(&point_light_vector, AP_VECTOR_LIGHT);
78 }
79
80 if (point_light_vector.length >= AP_LIGHT_POINT_NUM) {
81 LOGE("unable to generate new spot lights");
82 return AP_ERROR_INVALID_PARAMETER;
83 }
84
85 struct AP_Light light;
86 memset(&light, 0, sizeof(light));
87
88 light.type = AP_LIGHT_POINT;
89 ap_v3_copy(light.position, position);
90 ap_v3_copy(light.ambient, ambient);
91 ap_v3_copy(light.diffuse, diffuse);
92 ap_v3_copy(light.specular, specular);
93
94 memcpy(light.param, param, FLOAT_SIZE * AP_LIGHT_PARAM_NUM);
95
96 unsigned id = point_light_vector.length + 1;
97 light.id = id;
98 int ret = ap_vector_push_back(&point_light_vector, (char*) &light);
99 if (ret) {
100 LOGE("failed to generate light");
101 AP_CHECK(ret);
102 return AP_ERROR_INIT_FAILED;
103 }
104 *light_id = id;
105
106 return 0;
107}

109int ap_light_render()
110{
111 if (!point_light_vector.data) {
112 LOGE("failed to render lights: not initialized");
113 return AP_ERROR_INIT_FAILED;
114 }
115 char buffer[AP_DEFAULT_BUFFER_SIZE] = { 0 };
116
117 unsigned int old_shader = ap_get_current_shader();
118 unsigned int shader = 0;
119 ap_render_get_persp_shader(&shader);
120 if (shader == 0) {
121 LOGE("failed to render lights: renderer not initialized");
122 return AP_ERROR_INIT_FAILED;
123 }
124
125 ap_shader_use(shader);
126 struct AP_Light *light = (struct AP_Light*) point_light_vector.data;
127 int point_nr = 0;
128 // send point lights data
129 for (int i = 0; i < point_light_vector.length
130 && point_nr < AP_LIGHT_POINT_NUM; ++i)
131 {
132 struct AP_Light *p = light + i;
133
134 if (p->type != AP_LIGHT_POINT) {
135 continue;
136 }
137 // position
138 sprintf(buffer, AP_SP_PL_POSITION, i);
139 ap_shader_set_vec3(buffer, p->position);
140 // ambient
141 sprintf(buffer, AP_SP_PL_AMBIENT, i);
142 ap_shader_set_vec3(buffer, p->ambient);
143 // diffuse
144 sprintf(buffer, AP_SP_PL_DIFFUSE, i);
145 ap_shader_set_vec3(buffer, p->diffuse);
146 // specular
147 sprintf(buffer, AP_SP_PL_SPECULAR, i);
148 ap_shader_set_vec3(buffer, p->specular);
149 // constant
150 sprintf(buffer, AP_SP_PL_CONSTANT, i);
151 ap_shader_set_float(buffer, p->param[0]);
152 // linear
153 sprintf(buffer, AP_SP_PL_LINEAR, i);
154 ap_shader_set_float(buffer, p->param[1]);
155 // quadratic
156 sprintf(buffer, AP_SP_PL_QUADRATIC, i);
157 ap_shader_set_float(buffer, p->param[2]);
158 }
159
160 // send directional light data
161 if (direct_light.type == AP_LIGHT_DIRECTIONAL) {
162 // direction
163 ap_shader_set_vec3(AP_SP_DL_DIRECTION, direct_light.direction);
164 // ambient
165 ap_shader_set_vec3(AP_SP_DL_AMBIENT, direct_light.ambient);
166 // diffuse
167 ap_shader_set_vec3(AP_SP_DL_DIFFUSE, direct_light.diffuse);
168 // specular
169 ap_shader_set_vec3(AP_SP_DL_SPECULAR, direct_light.specular);
170 }
171

174 // ambient
175 ap_shader_set_vec3(AP_SP_SL_AMBIENT, spot_light.ambient);
176 // diffuse
177 ap_shader_set_vec3(AP_SP_SL_AMBIENT, spot_light.diffuse);
178 // specular
