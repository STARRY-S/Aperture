48int ap_shader_generate(
49 const char* vshader_path,
50 const char* fshader_path,
51 unsigned int *shader_id)
52{
53 if (!vshader_path || !fshader_path) {
54 return AP_ERROR_INVALID_PARAMETER;
55 }
56
57 // initialize vector when first use
58 if (shader_vector.data == NULL) {
59 ap_vector_init(&shader_vector, AP_VECTOR_SHADER);
60 }
61
62 GLuint gl_shader_id = 0;
63 gl_shader_id = ap_shader_load_program(vshader_path, fshader_path);
64 if (gl_shader_id == 0) {
65 return AP_ERROR_SHADER_LOAD_FAILED;
66 }
67 struct AP_Shader shader;
68 memset(&shader, 0, sizeof(struct AP_Shader));
69 shader.id = shader_vector.length + 1;
70 shader.opengl_program_id = gl_shader_id;
71 shader.fragment_shader_path = ap_char_copy(fshader_path);
72 shader.vertex_shader_path = ap_char_copy(vshader_path);
73 ap_vector_push_back(&shader_vector, (const char*) &shader);
74 *shader_id = shader.id;
75
76 return 0;
77}

148GLuint ap_compile_shader(
149 GLenum type,
150 const char *const shader_src)
151{
152 GLuint shader = 0;
153 GLint compiled = 0;
154
155 if (!(shader = glCreateShader(type))) {
156 LOGE("glCreateShader failed, type %d.", type);
157 return 0;
158 }
159
160 glShaderSource(shader, 1, &shader_src, NULL);
161 glCompileShader(shader);
162 glGetShaderiv(shader, GL_COMPILE_STATUS, &compiled);
163
164 if (!compiled) {
165 GLint info_len = 0;
166 glGetShaderiv(shader, GL_INFO_LOG_LENGTH, &info_len);
167 char *info = (char*) AP_MALLOC(info_len);
168 if (info == NULL) {
169 LOGE("ap_compile_shader: malloc error");
170 glDeleteShader(shader);
171 return 0;
172 }
173 glGetShaderInfoLog(shader, info_len, NULL, info);
174 LOGE("ap_compile_shader failed: \n%s", info);
175 AP_FREE(info);
176 return 0;
177 }
178 return shader;
179}

202GLuint ap_shader_load_program(
203 const char *const vshader_path,
204 const char *const fshader_path)
205{
206 GLuint vshader = ap_shader_load(GL_VERTEX_SHADER, vshader_path);
207 GLuint fshader = ap_shader_load(GL_FRAGMENT_SHADER, fshader_path);
208 if (!vshader || !fshader) {
209 return 0;
210 }
211 GLuint program = 0;
212 if (!(program = glCreateProgram())) {
213 LOGE("glCreateProgram failed.");
214 return 0;
215 }
216
217 glAttachShader(program, vshader);
218 glAttachShader(program, fshader);
219 glLinkProgram(program);
220 glDeleteShader(vshader);
221 glDeleteShader(fshader);
222
223 GLint linked = 0;
224 glGetProgramiv(program, GL_LINK_STATUS, &linked);
225 if (!linked) {
226 GLint log_length = 0;
227 glGetProgramiv(program, GL_INFO_LOG_LENGTH, &log_length);
228
229 GLchar *buffer = AP_MALLOC(sizeof(GLchar) * (log_length + 1));
230 glGetProgramInfoLog(program, log_length, &log_length, buffer);
231 LOGE("ap_shader_load_program: glLinkProgram\n%s", buffer);
232 glDeleteProgram(program);
233 return 0;
234 }
235 return program;
236}
