116int ap_model_generate(const char *path, unsigned int *model_id)
117{
118 if (!path || !model_id) {
119 return AP_ERROR_INVALID_PARAMETER;
120 }
121
122 // initialize vector when first use
123 if (model_vector.data == NULL) {
124 ap_vector_init(&model_vector, AP_VECTOR_MODEL);
125 }
126
127 struct AP_Model model;
128 AP_CHECK( ap_model_init_ptr(&model, path, false) );
129
130 model.id = model_vector.length + 1;
131 ap_vector_push_back(&model_vector, (const char*) &model);
132 *model_id = model.id;
133 LOGD("generated model %s id %d", path, *model_id);
134
135 return 0;
136}

138int ap_model_use(unsigned int model_id)

143

145 model_using = NULL;
146 return 0;

150 model_using = tmp_model + (model_id - 1);
151
152 return 0;

155int ap_model_draw()
156{
157 unsigned int old_shader = ap_get_current_shader();
158 unsigned int render_persp_shader_id = 0;
159 ap_render_get_persp_shader(&render_persp_shader_id);
160
161 if (model_using == NULL) {
162 return AP_ERROR_MODEL_NOT_SET;
163 }
164
165 if (render_persp_shader_id == 0) {
166 LOGE("failed to draw model: perspective shader is 0");
167 return AP_ERROR_SHADER_NOT_SET;
168 }
169
170 mat4 mat_model;
171 glm_mat4_identity(mat_model);
172 glm_scale(mat_model, model_using->scale);
173 glm_translate(mat_model, model_using->pos);
174 glm_rotate(mat_model, model_using->rotate_angle,
175 model_using->rotate_axis);
176 ap_render_set_model_mat((float *) mat_model);
177
178 ap_shader_use(render_persp_shader_id);
179 ap_model_draw_ptr_shader(model_using);
180 ap_shader_use(old_shader);
181
182 return 0;
183}

185int ap_model_free(int id)
186{
187 if (id <= 0) {
188 return AP_ERROR_INVALID_PARAMETER;
189 }
190 struct AP_Model *models = (struct AP_Model*) model_vector.data;
191 for (int i = 0; i < model_vector.length; ++i) {
192 if (models[i].id != id) {
193 continue;
194 }
195
196 AP_FREE(models[i].directory);
197 models[i].directory = NULL;
198
199 // Release all datas allocated in mesh
200 for (int j = 0; j < models->mesh_length; ++j) {
201 struct AP_Mesh *mesh = models[i].mesh + j;
202 if (mesh->VAO) {
203 glDeleteVertexArrays(1, &mesh->VAO);
204 }
205 if (mesh->VBO) {
206 glDeleteBuffers(1, &mesh->VBO);
207 }
208 if (mesh->EBO) {
209 glDeleteBuffers(1, &mesh->EBO);
210 }
211 if (mesh->indices) {
212 AP_FREE(mesh->indices);
213 mesh->indices = NULL;
214 }
215 mesh->indices_length = 0;
216 if (mesh->vertices) {
217 AP_FREE(mesh->vertices);
218 mesh->vertices = NULL;
219 }
220 mesh->vertices_length = 0;
221 if (mesh->textures) {
222 AP_FREE(mesh->textures);
223 mesh->textures = NULL;
224 }
225 mesh->texture_length = 0;
226 }
227 AP_FREE(models[i].mesh);
228 models[i].mesh = NULL;
229
230 AP_FREE(models[i].texture);
231 models[i].texture = NULL;
232 ap_vector_remove_data(
233 &model_vector, (char*) (models + i),
234 (char*) (models + i + 1), sizeof(struct AP_Model)
235 );
236 if (model_using == models + i) {
237 model_using = NULL;
238 }
239 }
240
241 return 0;

292static int ap_model_init_ptr(
293 struct AP_Model *model, const char *path, bool gamma)
294{
295 if (model == NULL) {
296 return AP_ERROR_INVALID_POINTER;
297 }
298
299 memset(model, 0, sizeof(struct AP_Model));
300 model->pos[0] = model->pos[1] = model->pos[2] = 0.0f;
301 model->scale[0] = model->scale[1] = model->scale[2] = 1.0f;
302 model->rotate_angle = 0.0f;
303 model->rotate_axis[0]
304 = model->rotate_axis[1]
305 = model->rotate_axis[2] = 0.0f;
306
307 int dir_char_pos = 0;
308 for (int i = strlen(path); i > 0; --i) {
309 if (path[i] == '/') {
310 dir_char_pos = i;
311 break;
312 }
313 }
314 if (dir_char_pos > 0) {
315 char *dir_path = AP_MALLOC(sizeof(char) * (dir_char_pos + 2));
316 memcpy(dir_path, path, (dir_char_pos + 1) * sizeof(char));
317 dir_path[dir_char_pos + 1] = '\0';
318 model->directory = dir_path;
319 } else if (dir_char_pos == 0) {
320 char *dir_path = AP_MALLOC(1 * sizeof(char));
321 dir_path[0] = '\0';
322 model->directory = dir_path;
323 }
324
325 return ap_model_load_ptr(model, path);
326}

328int ap_model_load_ptr(struct AP_Model *model, const char *path)
329{
330 if (model == NULL) {
331 return AP_ERROR_INVALID_POINTER;
332 }
333
334 // Start the import on the given file with some example postprocessing
335 // Usually if speed is not the most important aspect for you
336 // you'll probably to request more postprocessing
337 // than we do in this example.
338
339 // custom file io for assimp
340 struct aiFileIO fileIo;
341 fileIo.CloseProc = ap_custom_ai_file_close_proc;
342 fileIo.OpenProc = ap_custom_ai_file_open_proc;
343 fileIo.UserData = NULL;
344
345 const struct aiScene* scene = aiImportFileEx(
346 path,
347 aiProcess_Triangulate | aiProcess_GenSmoothNormals |
348 aiProcess_FlipUVs | aiProcess_CalcTangentSpace,
349 &fileIo
350 );
351
352 if(!scene || scene->mFlags & AI_SCENE_FLAGS_INCOMPLETE
353 || !scene->mRootNode)
354 {
355 LOGE("Assimp import failed: \n%s", aiGetErrorString());
356 return AP_ERROR_ASSIMP_IMPORT_FAILED;
357 }
358
359 // Now we can access the file's contents
360 AP_CHECK( ap_model_process_node(model, scene->mRootNode, scene) );
361
362 // We're done. Release all resources associated with this import
363 aiReleaseImport(scene);
364
365 return AP_ERROR_SUCCESS;
366}

368int ap_model_process_node(
369 struct AP_Model *model,
370 struct aiNode *node,
371 const struct aiScene *scene)
372{
373 if (model == NULL || node == NULL || scene == NULL) {
374 return AP_ERROR_INVALID_POINTER;
375 }
376 // process each mesh located at the current node
377 for (unsigned int i = 0; i < node->mNumMeshes; i++)
378 {
379 // the node object only contains indices to index the actual
380 // objects in the scene. the scene contains all the data,
381 // node is just to keep stuff organized
382 // (like relations between nodes).
383 struct aiMesh* mesh = scene->mMeshes[node->mMeshes[i]];
384 struct AP_Mesh *mesh_new =
385 ap_model_process_mesh(model, mesh, scene);
386 ap_model_mesh_push_back(model, mesh_new);
387 ap_mesh_free(mesh_new);
388 }
389
390 // after we've processed all of the meshes (if any)
391 // we then recursively process each of the children nodes
392 for (unsigned int i = 0; i < node->mNumChildren; i++)
393 {
394 ap_model_process_node(model, node->mChildren[i], scene);
395 }
396
397 return 0;
398}

400struct AP_Mesh *ap_model_process_mesh(struct AP_Model *model,
401 struct aiMesh *mesh,
402 const struct aiScene *scene)
403{
404 if (model == NULL || mesh == NULL || scene == NULL) {
405 return NULL;
406 }
407 // data to fill
408 struct AP_Vector vec_vertices;
409 struct AP_Vector vec_indices;
410 struct AP_Vector vec_textures;
411
412 ap_vector_init(&vec_vertices, AP_VECTOR_VERTEX);
413 ap_vector_init(&vec_indices, AP_VECTOR_UINT);
414 ap_vector_init(&vec_textures, AP_VECTOR_TEXTURE);
415
416 // walk through each of the mesh's vertices
417 for (unsigned int i = 0; i < mesh->mNumVertices; i++)
418 {
419 struct AP_Vertex vertex;
420 // positions
421 vertex.position[0] = mesh->mVertices[i].x;
422 vertex.position[1] = mesh->mVertices[i].y;
423 vertex.position[2] = mesh->mVertices[i].z;
424
425 // normals
426 if (mesh->mNormals != NULL) {
427 vertex.normal[0] = mesh->mNormals[i].x;
428 vertex.normal[1] = mesh->mNormals[i].y;
429 vertex.normal[2] = mesh->mNormals[i].z;
430 }
431 // texture coordinates
432 // does the mesh contain texture coordinates?
433 if(mesh->mTextureCoords[0]) {
439 vertex.tex_coords[0] = mesh->mTextureCoords[0][i].x;
440 vertex.tex_coords[1] = mesh->mTextureCoords[0][i].y;
441
442 // tangent
443 vertex.tangent[0] = mesh->mTangents[i].x;
444 vertex.tangent[1] = mesh->mTangents[i].y;
445 vertex.tangent[2] = mesh->mTangents[i].z;
446
447 // big tangent
448 vertex.big_tangent[0] = mesh->mBitangents[i].x;
449 vertex.big_tangent[1] = mesh->mBitangents[i].y;
450 vertex.big_tangent[2] = mesh->mBitangents[i].z;
451 } else {
452 vertex.tex_coords[0] = 0.0f;
453 vertex.tex_coords[1] = 0.0f;
454 }
455
456 // vertices.push_back(vertex);
457 ap_vector_push_back(&vec_vertices, (const char *) &vertex);
458 }
459 // now wak through each of the mesh's faces
460 // (a face is a mesh its triangle)
461 // and retrieve the corresponding vertex indices.
462 for (unsigned int i = 0; i < mesh->mNumFaces; i++)
463 {
464 struct aiFace face = mesh->mFaces[i];
465 // retrieve all indices of the face
466 // and store them in the indices vector
467 for(unsigned int j = 0; j < face.mNumIndices; j++) {
468 ap_vector_push_back(
469 &vec_indices, (const char *) &face.mIndices[j]
470 );
471 }
472 }
473 // process materials
474 struct aiMaterial* material = scene->mMaterials[mesh->mMaterialIndex];
486 // 1. diffuse maps
487 size_t size = sizeof(struct AP_Texture);
488 struct AP_Vector *vec_diffuse_ptr = NULL;
489 vec_diffuse_ptr = ap_model_load_material_textures(
490 model, material,
491 aiTextureType_DIFFUSE, AP_TEXTURE_TYPE_DIFFUSE
492 );
493 if (vec_diffuse_ptr->length > 0) {
494 ap_vector_insert_back(
495 &vec_textures,
496 vec_diffuse_ptr->data,
497 size * vec_diffuse_ptr->length
498 );
499 }
500 ap_vector_free(vec_diffuse_ptr);
501
502 // 2. specular maps
503 struct AP_Vector *vec_specular_ptr = NULL;
504 vec_specular_ptr = ap_model_load_material_textures(
505 model, material,
506 aiTextureType_SPECULAR, AP_TEXTURE_TYPE_SPECULAR
507 );
508 if (vec_specular_ptr->length > 0) {
509 ap_vector_insert_back(
510 &vec_textures,
511 vec_specular_ptr->data,
512 size * vec_specular_ptr->length
513 );
514 }
515 ap_vector_free(vec_specular_ptr);
516
517 // 3. normal maps
518 struct AP_Vector *vec_normal_ptr = NULL;
519 vec_normal_ptr = ap_model_load_material_textures(
520 model, material,
521 aiTextureType_NORMALS, AP_TEXTURE_TYPE_NORMAL
522 );
523 if (vec_normal_ptr->length > 0) {
524 ap_vector_insert_back(
525 &vec_textures,
526 vec_normal_ptr->data,
527 size * vec_normal_ptr->length
528 );
529 }
530 ap_vector_free(vec_normal_ptr);
531
532 // 4. height maps
533 struct AP_Vector *vec_height_ptr;
534 vec_height_ptr = ap_model_load_material_textures(
535 model, material,
536 aiTextureType_HEIGHT, AP_TEXTURE_TYPE_HEIGHT
537 );
538 if (vec_height_ptr->length > 0) {
539 ap_vector_insert_back(
540 &vec_textures,
541 vec_height_ptr->data,
542 size * vec_height_ptr->length
543 );
544 }
545 ap_vector_free(vec_height_ptr);
546
547 // return a mesh object created from the extracted mesh data
548 static struct AP_Mesh mesh_buffer;
549 ap_mesh_init(
550 &mesh_buffer,
551 (struct AP_Vertex *) vec_vertices.data,
552 vec_vertices.length,
553 (unsigned int *) vec_indices.data,
554 vec_indices.length,
555 (struct AP_Texture *) vec_textures.data,
556 vec_textures.length
557 );
558 ap_vector_free(&vec_vertices);
559 ap_vector_free(&vec_indices);
560 ap_vector_free(&vec_textures);
561
562 return &mesh_buffer;
563}

565struct AP_Vector *ap_model_load_material_textures(
566 struct AP_Model *model,
567 struct aiMaterial *mat,
568 enum aiTextureType type,
569 int ap_type)
570{
571 static struct AP_Vector vec_texture;
572 ap_vector_init(&vec_texture, AP_VECTOR_TEXTURE);
573
574 GLuint mat_texture_count = aiGetMaterialTextureCount(mat, type);
575 for (GLuint i = 0; i < mat_texture_count; i++)
576 {
577 struct aiString str;
578 aiGetMaterialTexture(
579 mat, type, i, &str,
580 NULL, NULL, NULL, NULL, NULL, NULL
581 );
582 struct AP_Texture *ptr = NULL;
583 ptr = ap_texture_get_ptr_by_path(str.data);
584 if (ptr == NULL) {
585 GLuint texture_id = 0;
586 ap_texture_generate(&texture_id, ap_type,
587 str.data, model->directory, false);
588 ptr = ap_texture_get_ptr(texture_id);
589 }
590 if (ptr) {
591 ap_vector_push_back(&vec_texture, (char*) ptr);
592 ap_model_texture_loaded_push_back(model, ptr);
593 }
594 }
595
596 struct aiColor4D ai_color = { 0.f, 0.f, 0.f, 0.f };
597 if (ap_type == AP_TEXTURE_TYPE_DIFFUSE) {
598 aiGetMaterialColor(mat, AI_MATKEY_COLOR_DIFFUSE, &ai_color);
599 } else if (ap_type == AP_TEXTURE_TYPE_SPECULAR) {
600 aiGetMaterialColor(mat, AI_MATKEY_COLOR_SPECULAR, &ai_color);
601 }
602
603 if (!(ai_color.r || ai_color.g || ai_color.b)) {
604 return &vec_texture;
605 }
606
607 unsigned int id = 0;
608 vec4 color = { 0.0f };
609 memcpy(color, &ai_color, sizeof(float) * 4);
610 struct AP_Texture *ptr = ap_texture_get_ptr_by_rgba(color);
611 if (ptr == NULL) {
612 ap_texture_generate_rgba(
613 &id, color, 16, AP_TEXTURE_TYPE_DIFFUSE
614 );
615 ptr = ap_texture_get_ptr(id);
616 }
617 if (ptr) {
618 ap_vector_push_back(&vec_texture, (char*) ptr);
619 ap_model_texture_loaded_push_back(model, ptr);
620 }
621
622 return &vec_texture;
623}

625int ap_model_texture_loaded_push_back(
626 struct AP_Model *model,
627 struct AP_Texture *texture)
628{
629 if (model == NULL || texture == NULL) {
630 return AP_ERROR_INVALID_PARAMETER;
631 }
632
633 // add a new texture struct object into model
634 model->texture = AP_REALLOC(model->texture,
635 sizeof(struct AP_Texture) * (model->texture_length + 1));
636 if (model->texture == NULL) {
637 LOGE("realloc error");
638 return AP_ERROR_MALLOC_FAILED;
639 }
640 struct AP_Texture *texture_new =
641 model->texture + (model->texture_length);
642 model->texture_length++;
643 ap_texture_init(texture_new);
644 texture_new->type = texture->type;
645 // Do not use texture_set_path here, just copy the path pointer to
646 // model->texture array.
647 texture_new->path = texture->path;
648 texture_new->id = texture->id;
649 memcpy(texture_new->RGBA, texture->RGBA, sizeof(float) * 4);
650
651 return 0;
652}
