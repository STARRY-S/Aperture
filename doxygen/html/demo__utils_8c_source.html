248int demo_render()
249{
250 ap_render_flush();
251 // render
252 glClearColor(0.1f, 0.1f, 0.1f, 1.0f);
253 glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
254
255 // depth test
256 glEnable(GL_BLEND);
257 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
258
259 // cull face
260 glEnable(GL_CULL_FACE);
261
262 // update player
263 ap_physic_update_creature();
264 // render the model
265 ap_model_use(model_id);
266 AP_CHECK( ap_model_draw() );
267
268 // render the lamp cube
269 ap_shader_use(cube_shader);
270 int view_distance = 0;
271 ap_render_get_view_distance(&view_distance);
272 ap_shader_set_float(
273 "view_distance",
274 (float) view_distance
275 );
276 glActiveTexture(GL_TEXTURE0);
277 glBindTexture(GL_TEXTURE_2D, light_texture);
278 float *projection = NULL;
279 ap_render_get_persp_matrix(&projection);
280 ap_shader_set_mat4("projection", projection);
281 float *view = NULL;
282 ap_render_get_view_matrix(&view);
283 ap_shader_set_mat4("view", view);
284
285 mat4 mat_model;
286 for (int i = 0; draw_light_cubes && i < DEMO_POINT_LIGHT_NUM; ++i) {
287 glm_mat4_identity(mat_model);
288 glm_translate(mat_model, light_positions[i]);
289 ap_shader_set_mat4("model", mat_model[0]);
290 glBindVertexArray(light_cube_VAO);
291 glDrawArrays(GL_TRIANGLES, 0, 36);
292 }
293
294 vec4 color = {0.9, 0.9, 0.9, 1.0};
295 int screen_height = ap_get_buffer_height();
296 float fps = 0;
297 ap_render_get_fps(&fps);
298 vec3 cam_position = { 0.0f, 0.0f, 0.0f };
299 vec3 cam_direction = { 0.0f, 0.0f, 0.0f };
300 ap_camera_get_position(cam_position);
301 ap_camera_get_front(cam_direction);
302 sprintf(buffer, "(%.1f, %.1f, %.1f) (%.1f, %.1f, %.1f) %4.1ffps",
303 player->box.pos[0],
304 player->box.pos[1] - (player->box.size[1] / 2),
305 player->box.pos[2],
306 cam_direction[0], cam_direction[1], cam_direction[2], fps);
307 ap_render_text_line(buffer, 10.0, screen_height - 30.0, 1.0, color);
308 sprintf(buffer, "float: %d (%.1f, %.1f, %.1f)",
309 player->floating, player->move.speed[0], player->move.speed[1],
310 player->move.speed[2]);
311 ap_render_text_line(buffer, 10.0, screen_height - 60.0, 1.0, color);
312 ap_render_aim_dot();
313 ap_render_aim_cross();
314
315 glBindVertexArray(0);
316 ap_shader_use(0);
317 glDisable(GL_CULL_FACE);
318 glDisable(GL_BLEND);
319
320 return EXIT_SUCCESS;
321}
