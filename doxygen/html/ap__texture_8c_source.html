151unsigned int ap_texture_from_file(
152 const char *path,
153 const char *directory,
154 bool gamma)
155{
156 if (!path || !directory) {
157 LOGE("ap_texture_from_file: path %p, dir %p", path, directory);
158 return 0;
159 }
160 // stbi_set_flip_vertically_on_load(true);
161 unsigned int texture_id;
162 glGenTextures(1, &texture_id);
163 if (texture_id == 0) {
164 LOGW("glGenTextures failed");
165 }
166
167 int width, height, nr_components;
168 int buffer_length = strlen(path) + strlen(directory) + 1;
169 char *path_buffer = AP_MALLOC(sizeof(char) * buffer_length);
170 if (path_buffer == NULL) {
171 LOGE("malloc failed");
172 return 0;
173 }
174 sprintf(path_buffer, "%s%s", directory, path);
175
176 int length = 0;
177 unsigned char *buffer = NULL;
178 ap_read_file_to_memory(path_buffer, (char**) &buffer, &length);
179 unsigned char *data = stbi_load_from_memory(
180 buffer, length, &width, &height, &nr_components, 0
181 );
182 AP_FREE(buffer);
183 buffer = NULL;
184
185 if (!data) {
186 LOGE("Failed to load texture: %s", path_buffer);
187 glDeleteTextures(1, &texture_id);
188 AP_FREE(path_buffer);
189 path_buffer = NULL;
190 return 0;
191 }
192
193 int format = 0;
194 if (nr_components == 1) {
195 format = GL_RED;
196 } else if (nr_components == 3) {
197 format = GL_RGB;
198 } else if (nr_components == 4) {
199 format = GL_RGBA;
200 }
201
202 glBindTexture(GL_TEXTURE_2D, texture_id);
203 glTexImage2D(
204 GL_TEXTURE_2D, 0, format, width, height, 0,
205 format, GL_UNSIGNED_BYTE, data
206 );
207 glGenerateMipmap(GL_TEXTURE_2D);
208
209 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
210 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
211 glTexParameteri(
212 GL_TEXTURE_2D,
213 GL_TEXTURE_MIN_FILTER,
214 GL_NEAREST_MIPMAP_NEAREST
215 );
216 glTexParameteri(
217 GL_TEXTURE_2D,
218 GL_TEXTURE_MAG_FILTER,
219 GL_NEAREST
220 );
221
222 stbi_image_free(data);
223 AP_FREE(path_buffer);
224 path_buffer = NULL;
225 return texture_id;
226}

228unsigned int ap_texture_from_rgba(vec4 color, int size)
229{
230 unsigned char* data = (unsigned char*) AP_MALLOC(
231 4 * size * size * sizeof(unsigned char)
232 );
233 memset(data, 0, 4 * size * size * sizeof(unsigned char));
234 if (data == NULL) {
235 LOGE("malloc failed");
236 return 0;
237 }
238
239 unsigned int texture = 0;
240 glGenTextures(1, &texture);
241 glBindTexture(GL_TEXTURE_2D, texture);
242
243 // set texture wrap parameters
244 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
245 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
246 // set texture filter parameters
247 glTexParameteri(
248 GL_TEXTURE_2D,
249 GL_TEXTURE_MIN_FILTER,
250 GL_NEAREST_MIPMAP_NEAREST
251 );
252 glTexParameteri(
253 GL_TEXTURE_2D,
254 GL_TEXTURE_MAG_FILTER,
255 GL_NEAREST
256 );
257
258 for (int i = 0; i < (size * size); ++i) {
259 for (int j = 0; j < 4; ++j) {
260 data[i * 4 + j] = (color[j] * 255);
261 }
262 }
263 glTexImage2D(
264 GL_TEXTURE_2D, 0, GL_RGBA, size, size,
265 0, GL_RGBA, GL_UNSIGNED_BYTE, data
266 );
267 glGenerateMipmap(GL_TEXTURE_2D);
268 glBindTexture(GL_TEXTURE_2D, 0);
269
270 AP_FREE(data);
271
272 return texture;
273}
