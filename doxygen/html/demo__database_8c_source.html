23static int demo_init_creature_cb(void *r, int argc, char **argv, char **name)
24{
25 if (argc == 0) {
26 db_has_data = false;
27 return 0;
28 }
29
30 vec3 pos = {0};
31 vec2 euler = {0};
32 for (int i = 0; i < argc; ++i) {
33 // LOGD("%d: %s = %s", i, name[i], argv[i]);
34 if (!argv[i]) {
35 continue;
36 }
37 if (strcmp(name[i], "POS_X") == 0) {
38 pos[0] = (float) ap_c2d(argv[i]);
39 } else if (strcmp(name[i], "POS_Y") == 0) {
40 pos[1] = (float) ap_c2d(argv[i]);
41 } else if (strcmp(name[i], "POS_Z") == 0) {
42 pos[2] = (float) ap_c2d(argv[i]);
43 } else if (strcmp(name[i], "YAW") == 0) {
44 euler[0] = (float) ap_c2d(argv[i]);
45 } else if (strcmp(name[i], "PITCH") == 0) {
46 euler[1] = (float) ap_c2d(argv[i]);
47 }
48 }
49
50 LOGD("restore position (%.3f, %.3f, %.3f)", pos[0], pos[1], pos[2]);
51 LOGD("restore Euler YAW = %.3f, PITCH = %.3f",
52 euler[0], euler[1]);
53 memcpy(db_restore_creature_pos, pos, VEC3_SIZE);
54 memcpy(db_restore_creature_euler, euler, VEC2_SIZE);
55 db_has_data = true;
56
57 return 0;
58}
